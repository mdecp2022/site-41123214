# 導入亂數模組
import random
# 導入字串模組
import string
# 利用 def 關鍵字定義函式
# 函式的輸入變數可以設定初始值
def password_generator(size=4, chars=string.ascii_lowercase + string.digits):
# 函式內以多行註解說明函式功能
"""Generate random password
"""
# 利用 return 關鍵字將所產生的亂數字串傳回
return ''.join(random.choice(chars) for _ in range(size))
# 建立一個變數與隨後的字串對應
